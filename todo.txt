# Create 'category' in database.  A category must be selected when a transaction is entered.

{
  id: integer auto increment 
  master_category_id: integer foreign key

  sort_order: integer
  create_date: timestamp with time zone default current_timestamp
  update_date: timestamp with time zone
}
